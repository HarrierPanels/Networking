sudo nano /etc/netplan/01-network-manager-all.yaml

network:
  version: 2
  renderer: NetworkManager
  ethernets:
    enp0s3:
      routes:
        - to: 192.168.0.1 #default
          via: 10.2.73.254
        - to: 172.17.24.0/24
          via: 10.2.73.254
        - to: default
          via: 172.16.14.2
      nameservers:
        addresses: [192.168.0.1]
      dhcp4: yes
    enp0s8:
      addresses: [172.16.14.2/24]
      dhcp4: no
      routes:
        - to: 172.17.34.0/24
          via: 172.16.14.2


sudo netplan apply
